<script>
    const canvas = document.getElementById('stars');
    const ctx = canvas.getContext('2d');
    canvas.width = window.innerWidth; canvas.height = window.innerHeight;
    let stars = [];
    let isHeartMode = false;

    for(let i=0; i<100; i++) {
        stars.push({ x: Math.random()*canvas.width, y: Math.random()*canvas.height, size: Math.random()*2 + 1, speed: Math.random()*0.5 + 0.2 });
    }

    function drawHeart(x, y) {
        ctx.beginPath();
        ctx.moveTo(x, y);
        ctx.bezierCurveTo(x, y - 3, x - 5, y - 3, x - 5, y);
        ctx.bezierCurveTo(x - 5, y + 3, x, y + 5, x, y + 9);
        ctx.bezierCurveTo(x, y + 5, x + 5, y + 3, x + 5, y);
        ctx.bezierCurveTo(x + 5, y - 3, x, y - 3, x, y);
        ctx.fill();
    }
    
    function animate() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ctx.fillStyle = isHeartMode ? "rgba(255, 0, 85, 0.8)" : "white";
        stars.forEach(s => {
            if (isHeartMode) { drawHeart(s.x, s.y); } 
            else { ctx.beginPath(); ctx.arc(s.x, s.y, s.size, 0, Math.PI*2); ctx.fill(); }
            s.y -= s.speed;
            if(s.y < 0) s.y = canvas.height;
        });
        requestAnimationFrame(animate);
    }
    animate();

    function startJourney() {
        document.getElementById('mainMusic').play().catch(() => {});
        next(1);
    }

    function next(num) {
        const current = document.getElementById('s' + num);
        const following = document.getElementById('s' + (num + 1));
        if (num === 8) isHeartMode = true; 
        current.style.opacity = '0';
        setTimeout(() => {
            current.classList.add('hidden');
            current.classList.remove('visible');
            following.classList.remove('hidden');
            following.classList.add('visible');
        }, 600);
    }

    function checkPass() {
        const val = document.getElementById('passInput').value.toLowerCase();
        if(val.includes("i love you too")) { next(7); } 
        else { alert("Try again, Billu! ❤️"); }
    }
</script>
